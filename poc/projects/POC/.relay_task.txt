def _read_artifact(path: str) -> str:
    """Read artifact up to MAX_ARTIFACT_CHARS. Returns '' on empty path or any error."""
    if not path:
        return ''
    try:
        with open(path) as f:
            return f.read(MAX_ARTIFACT_CHARS)
    except OSError:
        return ''

def _extract_tokens(text: str) -> set:
    """Simple tokenization: lowercase, split on non-alpha, drop short tokens."""
    import re
    return {t for t in re.split(r'[^a-z]+', text.lower()) if len(t) >= 4}

def _mean_artifact_length(entry: ConfidenceEntry) -> float:
    """Mean of stored artifact lengths. 0.0 if no history."""
    lengths = getattr(entry, 'artifact_lengths', [])
    return sum(lengths) / len(lengths) if lengths else 0.0

def _check_content_novelty(artifact_text: str, entry: ConfidenceEntry) -> tuple:
    """Run Phase 1 + Phase 2 content checks. Returns (fired: bool, reason: str)."""
    # Phase 1a — length anomaly
    mean_len = _mean_artifact_length(entry)
    if mean_len > 0:
        ratio = len(artifact_text) / mean_len
        if ratio < ARTIFACT_LENGTH_RATIO_LOW:
            return True, (
                f"Artifact is unusually short ({len(artifact_text)} chars, "
                f"{ratio:.1%} of mean {mean_len:.0f})"
            )
        if ratio > ARTIFACT_LENGTH_RATIO_HIGH:
            return True, (
                f"Artifact is unusually long ({len(artifact_text)} chars, "
                f"{ratio:.1%} of mean {mean_len:.0f})"
            )

    # Phase 1b — correction pattern match
    # Differentials AND conversation texts are co-equal inputs to the same token extraction
    differentials = getattr(entry, 'differentials', [])
    correction_tokens = set()
    for d in differentials:
        if d.get('outcome') in ('correct', 'reject'):
            correction_tokens |= _extract_tokens(d.get('summary', ''))
    for conv_text in getattr(entry, 'conversations', []):
        correction_tokens |= _extract_tokens(conv_text)

    artifact_tokens = _extract_tokens(artifact_text)
    matched = correction_tokens & artifact_tokens
    if matched:
        sample = sorted(matched)[:3]
        return True, f"Artifact matches past correction patterns: {sample}"

    # Phase 2 — missing expected content
    from collections import Counter
    token_counts: Counter = Counter()
    for d in differentials:
        for tok in _extract_tokens(d.get('summary', '')):
            token_counts[tok] += 1
    for conv_text in getattr(entry, 'conversations', []):
        for tok in _extract_tokens(conv_text):
            token_counts[tok] += 1
    expected = {tok for tok, cnt in token_counts.items()
                if cnt >= EXPECTATION_MIN_DIFFERENTIALS}
    missing = expected - artifact_tokens
    if missing:
        sample = sorted(missing)[:3]
        return True, f"Artifact missing expected content from correction history: {sample}"

    return False, ''

    local state="$1" outcome="$2"

    local artifact="${4:-}"
    local conversation_text="${5:-}"

    if [[ -n "$PROXY_MODEL" ]]; then
      local diff_args=()

      local artifact_args=()

        --record --state "$state" --task-type "$task_type" \
        --outcome "$outcome" \
        ${diff_args[@]+"${diff_args[@]}"} \
        ${artifact_args[@]+"${artifact_args[@]}"} \
        ${conv_args[@]+"${conv_args[@]}"} \
        --model "$PROXY_MODEL" 2>/dev/null || true
    fi
  }
